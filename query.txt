1. Seleziona tutti gli utenti e calcolane l'et√† (25)
SELECT TIMESTAMPDIFF(YEAR, birthdate, CURDATE()) AS eta FROM users;


2. Seleziona tutti i post senza Like (13)
SELECT posts.* FROM posts
LEFT JOIN likes ON posts.id = likes.post_id WHERE likes.post_id IS NULL;


3. Conta il numero di like per ogni post (165)
SELECT posts.id AS post_id, COUNT(likes.post_id) AS like_quantity FROM posts
LEFT JOIN likes ON posts.id = likes.post_id GROUP BY posts.id;


4. Ordina gli utenti per il numero di media caricati (25)
SELECT users.*, COUNT(medias.user_id) AS media_quantity FROM users
LEFT JOIN medias ON users.id = medias.user_id
GROUP BY users.id ORDER BY media_quantity;


5. Ordina gli utenti per totale di likes ricevuti nei loro posts (25)
SELECT users.*, COUNT(likes.user_id) AS total_likes FROM users
LEFT JOIN posts ON users.id = posts.user_id
LEFT JOIN likes ON posts.id = likes.post_id
GROUP BY users.id ORDER BY total_likes;


6. Seleziona tutti i post degli utenti tra i 20 e i 30 anni (49)
SELECT posts.* FROM posts
LEFT JOIN users ON users.id = posts.user_id
WHERE TIMESTAMPDIFF(YEAR, birthdate, CURDATE()) BETWEEN 20 AND 30;


7. Seleziona il numero di post e di media per ogni utente (25)
SELECT users.*, posts.post_quantity AS post_quantity, medias.media_quantity AS media_quantity FROM users
LEFT JOIN (SELECT user_id, COUNT(*) AS post_quantity FROM posts GROUP BY user_id) posts
    ON users.id = posts.user_id
LEFT JOIN (SELECT user_id, COUNT(*) AS media_quantity FROM medias GROUP BY user_id) medias
    ON users.id = medias.user_id;

SELECT *, (LENGTH(tags) - LENGTH(REPLACE(tags, ',', '')) + 1) AS tags_count FROM posts

10. Ordina gli utenti in base al numero di tag usati nei loro post (25)
SELECT users.*, SUM(LENGTH(posts.tags) - LENGTH(REPLACE(posts.tags, ',', '')) + 1) AS tags_count FROM users
LEFT JOIN posts ON users.id = posts.user_id
GROUP BY users.id ORDER BY tags_count;
